3) Database schema (run in Supabase SQL Editor)
------------------------------------------------------------
Create tables:

-- journal posts
create table if not exists public.journal_posts (
  id text primary key,
  category text not null,
  title text not null,
  body text not null,
  created_at timestamptz not null default now(),
  updated_at timestamptz not null default now(),
  author_id uuid null
);

-- keep updated_at current
create or replace function public.set_updated_at()
returns trigger
language plpgsql
as $$
begin
  new.updated_at = now();
  return new;
end;
$$;

drop trigger if exists journal_posts_set_updated_at on public.journal_posts;
create trigger journal_posts_set_updated_at
before update on public.journal_posts
for each row
execute function public.set_updated_at();

-- post images
create table if not exists public.journal_post_images (
  id bigserial primary key,
  post_id text not null references public.journal_posts(id) on delete cascade,
  path text not null,
  sort_order int not null default 0,
  created_at timestamptz not null default now()
);

create index if not exists journal_post_images_post_id_idx
  on public.journal_post_images(post_id);

4) Storage bucket for images (Dashboard)
------------------------------------------------------------
- Storage -> Create bucket
- Bucket name: journal-images

IMPORTANT (matches current code): set the bucket to PUBLIC for read.
The current /journal page uses getPublicUrl(), so images must be publicly readable.

Storage policies (required for uploads from /admin)
------------------------------------------------------------
Supabase Storage uses the table: storage.objects

In SQL Editor, run:

-- Allow public read of objects in journal-images (optional if bucket is PUBLIC, but safe to add)
create policy "journal_images_public_read"
on storage.objects
for select
to anon, authenticated
using (bucket_id = 'journal-images');

-- Allow authenticated uploads to journal-images
create policy "journal_images_auth_insert"
on storage.objects
for insert
to authenticated
with check (bucket_id = 'journal-images');

-- Allow authenticated deletes in journal-images
create policy "journal_images_auth_delete"
on storage.objects
for delete
to authenticated
using (bucket_id = 'journal-images');

NOTE: If you use the optional "admin role" setup (profiles table), you should replace
the authenticated insert/delete policies with admin-only checks.

------------------------------------------------------------
5) RLS policies (Admin-only writes)
------------------------------------------------------------
Enable RLS:

alter table public.journal_posts enable row level security;
alter table public.journal_post_images enable row level security;

Recommended policy setup (works with email+password + NextAuth allowlist):

A) Allow PUBLIC READ (so /journal works without login)
create policy "journal_posts_read" on public.journal_posts
for select
to anon, authenticated
using (true);

create policy "journal_post_images_read" on public.journal_post_images
for select
to anon, authenticated
using (true);

B) Allow WRITES ONLY for authenticated users
NOTE: This is still enforced by NextAuth allowlist in the app.
If you want DB-level admin-only enforcement, see section C.

create policy "journal_posts_write_auth" on public.journal_posts
for insert
to authenticated
with check (true);

create policy "journal_posts_update_auth" on public.journal_posts
for update
to authenticated
using (true)
with check (true);

create policy "journal_posts_delete_auth" on public.journal_posts
for delete
to authenticated
using (true);

create policy "journal_post_images_write_auth" on public.journal_post_images
for insert
to authenticated
with check (true);

create policy "journal_post_images_delete_auth" on public.journal_post_images
for delete
to authenticated
using (true);

C) Optional: Strict admin-only writes at DB level (recommended)
1) Create a profiles table:
create table if not exists public.profiles (
  id uuid primary key references auth.users(id) on delete cascade,
  role text not null default 'user'
);

alter table public.profiles enable row level security;

create policy "profiles_read_own" on public.profiles
for select
to authenticated
using (auth.uid() = id);

create policy "profiles_upsert_own" on public.profiles
for insert
to authenticated
with check (auth.uid() = id);

create policy "profiles_update_own" on public.profiles
for update
to authenticated
using (auth.uid() = id)
with check (auth.uid() = id);

2) Set your admin role (run once, replace <ADMIN_USER_UUID>):
insert into public.profiles (id, role)
values ('<ADMIN_USER_UUID>', 'admin')
on conflict (id) do update set role = 'admin';

3) Update write policies to require admin role:
-- Example for journal_posts insert:
drop policy if exists "journal_posts_write_auth" on public.journal_posts;
create policy "journal_posts_write_admin" on public.journal_posts
for insert
to authenticated
with check (exists (
  select 1 from public.profiles p
  where p.id = auth.uid() and p.role = 'admin'
));

Repeat similarly for update/delete and journal_post_images.
